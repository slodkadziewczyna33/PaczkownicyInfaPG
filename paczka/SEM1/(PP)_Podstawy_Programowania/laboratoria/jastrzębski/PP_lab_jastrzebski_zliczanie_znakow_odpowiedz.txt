#include <iostream>

int licz_znak(const char* napis, const char znak){
    if(*napis=='\0')
        return 0;
    if(*napis==znak)
        return 1+licz_znak(napis+1, znak);
    else
        return licz_znak(napis+1, znak);
}

}
